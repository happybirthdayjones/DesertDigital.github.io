#Note Main.cpp is the class with the main method

#Also Note Hash.cpp contains a class with a header file

Name the makefile "Makefile" and use the command "make" to execute the makefile

Main.o: Main.cpp

Hash.o: Hash.cpp Hash.h

#Clears all object and output files
clean:
    rm *.o output
